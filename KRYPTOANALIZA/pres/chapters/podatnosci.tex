\subsection{Złamanie A5/1}
\begin{frame}{Budowa A5/1}
	\begin{itemize}
		\item Oparty na trzech rejestrach przesuwających (LFSR)
		\item IV:
		      \begin{itemize}
			      \item 64 bity klucza
			      \item 22 bity numeru ramki
		      \end{itemize}
		\item Stan: 19 + 22 + 23 = 64 bity
		\item Keystream: 2*114 bitów
	\end{itemize}

	\begin{figure}
		\centering
		\includegraphics[width=0.5\linewidth]{images/A5-1_GSM_cipher.svg.png}
	\end{figure}
\end{frame}

\begin{frame}{Złamanie A5/1}
	Offline:
	\begin{itemize}
		\item Mamy \textbf{$2^{64+22}$} możliwych kluczy \pause
		\item Ale tylko \textbf{$2^{64}$} możliwych stanów (tak naprawdę $2^{61.16}$) \pause
		\item Zamiast pamiętać $2^{64}$ stanów, używamy \textbf{Rainbow Tables} \pause
	\end{itemize}

	Online:
	\begin{itemize}
		\item Przechwytujemy szyfrogram
		\item Odzyskujemy keystream dzięki znanym tekstom jawnym (np. nagłówkom)
		\item Przeszukujemy Rainbow Tables, aby znaleźć odpowiadający stan i klucz
	\end{itemize}
\end{frame}

\begin{frame}{Downgrade Attack}
	\begin{itemize}
		\item Stawiamy fałszywą \textbf{stację bazową}, która jest silniejsza od prawdziwej
		\item Obsługujemy tylko \textbf{2G} oraz \textbf{A5/2} (A5/1 z krótszym kluczem)
	\end{itemize}
\end{frame}

\begin{frame}{IMSI Catcher (Stingray)}
	\begin{itemize}
		\item Urządzenie podszywające się pod stację bazową (około \$100k)
		\item Pozyskuje \textbf{IMSI} (International Mobile Subscriber Identity)
		\item Oraz \textbf{IMEI} (International Mobile Equipment Identity)
	\end{itemize}

\end{frame}
